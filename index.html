Typeflows DSL

class Deploy : WorkflowBuilder {
    override fun toWorkflow() = Workflow("Deploy to Production") {
        on += Push {
            branches = Branches.Only("main")
            paths = Paths.Only("src/**")
        }

        val buildJob = Job("build", UBUNTU_LATEST) {
            steps += checkout()

            steps += UseAction("actions/setup-node@v4") {
                with["node-version"] = "20"
            }

            steps += RunCommand("npm run build && npm test")
        }

        jobs += buildJob

        jobs += Job("deploy", UBUNTU_LATEST) {
            needs += buildJob

            steps += UseAction("actions/deploy@v2") {
                with["target"] = "production"
                with["token"] = "${{ secrets.DEPLOY_TOKEN }}"
            }
        }
    }
}

public class Deploy implements WorkflowBuilder {

    @NotNull
    @Override
    public Workflow toWorkflow() {
        return Workflow.configure("Deploy to Production", workflow -> {
            workflow.on.add(Push.configure(push -> {
                push.branches = Branches.Only("main");
                push.paths = Paths.Only("src/**");
            }));

            Job buildJob = Job.configure("build", UBUNTU_LATEST, job -> {
                job.steps.add(checkout());

                job.steps.add(UseAction.configure("actions/setup-node@v4", "node", action -> {
                    action.with.put("node-version", "20");
                }));

                job.steps.add(RunCommand.configure("npm run build && npm test"));
            });

            workflow.jobs.add(buildJob);

            workflow.jobs.add(Job.configure("deploy", UBUNTU_LATEST, job -> {
                job.needs.add(buildJob);
                job.steps.add(UseAction.configure("actions/deploy@v2", "deploy", action -> {
                    action.with.put("target", "production");
                    action.with.put("token", "${{ secrets.DEPLOY_TOKEN }}");
                }));
            }));
        });
    }
}

class Deploy(WorkflowBuilder):
    def to_workflow(self):
        with Workflow("Deploy to Production") as workflow:
            workflow.on += Push(
                branches=Branches.only("main"),
                paths=Paths.only("src/**")
            )

            build_job = Job("build", UBUNTU_LATEST)
            with build_job:
                build_job.steps += checkout()

                build_job.steps += use_action("actions/setup-node@v4",
                                              **{"node-version": "20"}
                                              )

                build_job.steps += run_command("npm run build && npm test")

            workflow.jobs += build_job

            with Job("deploy", UBUNTU_LATEST) as deploy_job:
                deploy_job.needs += build_job

                deploy_job.steps += use_action("actions/deploy@v2",
                                               target="production",
                                               token="${{ secrets.DEPLOY_TOKEN }}"
                                               )

            workflow.jobs += deploy_job

        return workflow

Generated YAML

.github/workflows/Deploy.yml 
name: Deploy to Production

on:
  push:
    branches: [main]
    paths: ['src/**']

jobs:
  build:
    runs-on: ubuntu-22.04
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: '20'
      - name: Build and Test
        run: npm run build && npm test

  deploy:
    needs: [build]
    runs-on: ubuntu-latest
    environment: production
    steps:
      - uses: actions/deploy@v2
        with:
          target: production
          token: ${{ secrets.DEPLOY_TOKEN }}

Common questions about adopting Typeflows Workflows-as-Code

Typeflows is a developer tool that runs locally, not a SaaS. It consists of:

• Language-specific libraries (JVM (Kotlin/Java), TypeScript, Python +others) that you install via your package manager (Gradle, npm, pip)
• CLI tools for project setup, workflow generation, and testing
• Build integrations that plug into your existing development workflow

Everything runs on your machines and CI environment - no data is sent to external services. You maintain full control over your code and workflows.
