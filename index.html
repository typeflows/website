class CI : Builder<Workflow> {

	override fun build() = Workflow("Build") {
		on += WorkflowDispatch.configure()

		permissions = Permissions(Contents to Write)

		on += Push {
			branches = Branches.Main
			paths = Paths.Ignore("**/.md")
		}

		on += PullRequest {
			paths = Paths.Ignore("**/.md")
		}

		jobs += Job("build", UBUNTU_LATEST) {
			steps += Checkout()

			steps += SetupJava(Adopt, V21)

			steps += SetupGradle()

			steps += RunCommand("./gradlew check")

			steps += UseAction(
				"mikepenz/action-junit-report@v5.6.2"
			) {
				condition = always()
				with["report_paths"] = "**/TEST-*.xml"
				with["github_token"] = GITHUB_TOKEN
			}
		}
	}
}

public class CI implements Builder<Workflow> {

	@NotNull
	@Override
	public Workflow build() {
		return Workflow.configure("build", wf -> {
			wf.displayName = "Build";
			wf.on.add(WorkflowDispatch.configure());

			wf.permissions = Permissions.configure(
				new HashMap<>() {{
					put(Contents, Write);
				}}
			);

			wf.on.add(Push.configure(push -> {
				push.branches = Branches.Main;
				push.paths = Paths.Ignore("**/.md");
			}));

			wf.on.add(PullRequest.configure(pr -> {
				pr.paths = Paths.Ignore("**/.md");
			}));

			wf.jobs.add(Job.configure("build",
				UBUNTU_LATEST, job -> {
					job.steps.add(Checkout.configure());

					job.steps.add(
					    SetupJava.configure(Adopt, V21)
					);

					job.steps.add(SetupGradle.configure());

					job.steps.add(
					    RunCommand.configure("./gradlew check")
					);

					job.steps.add(UseAction.configure(
					    "mikepenz/action-junit-report@v5.6.2", action -> {
					        action.condition = always();
					        action.with.put("report_paths", "**/TEST-*.xml");
					        action.with.put("github_token", GITHUB_TOKEN);
					    }
					));
				}));
		});
	}
}

class CI(Builder[Workflow]):
	def build(self) -> Workflow:
		with Workflow.configure("Build") as workflow:
			workflow.on.add(WorkflowDispatch.configure())

			workflow.permissions = Permissions.configure({
				Contents: Write
			})

			with Push.configure() as push:
				push.branches = Branches.Main
				push.paths = Paths.ignore("**/.md")

			workflow.on.add(push)

			with PullRequest.configure() as pr:
				pr.paths = Paths.ignore("**/.md")

			workflow.on.add(pr)

			with Job.configure("build", UBUNTU_LATEST) as job:
				job.steps.add(Checkout.configure())

				job.steps.add(SetupJava.configure(Adopt, V21))

				job.steps.add(SetupGradle.configure())

				job.steps.add(RunCommand.configure("./gradlew check"))

				with UseAction.configure(
					"mikepenz/action-junit-report@v5.6.2"
				) as action:
					action.condition = "always()"
					action.with_["report_paths"] = "**/TEST-*.xml"
					action.with_["github_token"] = GITHUB_TOKEN

				job.steps.add(action)

			workflow.jobs.add(job)

		return workflow

Compose all components into a single DotGitHub class that manages your entire .github/ folder. One class, complete control.

class TypeflowsRepo : Builder<TypeflowsGitHubRepo> {

	override fun build() = TypeflowsGitHubRepo {

		dotGithub = DotGitHub {
			workflows += CI()
			workflows += Deploy()

			files += StandardDependabot()

			files += StandardSecretScanning()

			files += TeamCodeOwners("myteam")

			files += CopilotInstructions {
				content +=
					markdownMd("hexagonal-architecture.md") +
					    markdownMd("tdd.md") +
					    markdownMd("kotlin-style.md")

				content += Microservice()
			}
		}

	}

	private fun markdownMd(resourceName: String) =
		fromResource(example.java.Microservice::class.java, resourceName)
}

public class TypeflowsRepo implements Builder<TypeflowsGitHubRepo> {

	@Override
	public TypeflowsGitHubRepo build() {
		return TypeflowsGitHubRepo.configure(repo ->

			repo.dotGithub = DotGitHub.configure(dsl -> {
				dsl.workflows.add(new CI());
				dsl.workflows.add(new Deploy());

				dsl.files.add(new StandardDependabot());

				dsl.files.add(new StandardSecretScanning());

				dsl.files.add(new TeamCodeOwners("myteam"));

				dsl.files.add(CopilotInstructions.configure(copilot ->
					copilot.content = copilot.content
					    .plus(markdownMd("java-style.md"))
					    .plus(new Microservice()))
				);
			})
		);
	}

	@NotNull
	private static MarkdownContent markdownMd(String resourceName) {
		return fromResource(TypeflowsRepo.class, resourceName);
	}
}

export class TypeflowsRepo implements Builder<TypeflowsRepo> {

class TypeflowsRepo(Builder[TypeflowsGitHubRepo]):

	def build(self) -> TypeflowsGitHubRepo:

		with TypeflowsGitHubRepo() as repo:

			with DotGitHub() as dsl:
				dsl.workflows.append(CI())
				dsl.workflows.append(Deploy())

				dsl.files.append(StandardDependabot())

				dsl.files.append(StandardSecretScanning())
				dsl.files.append(TeamCodeOwners("myteam"))

				with CopilotInstructions() as copilot:
					copilot.content = copilot.content.plus(
					    MarkdownContent.from_resource("python-style.md")
					).plus(
					    Microservice()
					)
				dsl.files.append(copilot)
			repo.dotGitHub = dsl

		return repo

Typeflows DSL

class Deploy : Builder<Workflow> {
	override fun build() = Workflow("Deploy to Production") {
		on += Push {
			branches = Branches.Only("main")
			paths = Paths.Only("src/**")
		}

		val buildJob = Job("build", UBUNTU_LATEST) {
			steps += Checkout()

			steps += UseAction("actions/setup-node@v4") {
				with["node-version"] = "20"
			}

			steps += RunCommand("npm run build && npm test")
		}

		jobs += buildJob

		jobs += Job("deploy", UBUNTU_LATEST) {
			needs += buildJob

			steps += UseAction("actions/deploy@v2") {
				with["target"] = "production"
				with["token"] = $$"${{ secrets.DEPLOY_TOKEN }}"
			}
		}
	}
}

public class Deploy implements Builder<Workflow> {

	@NotNull
	@Override
	public Workflow build() {
		return Workflow.configure("deploy-to-prod", workflow -> {
			workflow.displayName = "Deploy to Production";
			workflow.on.add(Push.configure(push -> {
				push.branches = Branches.Only("main");
				push.paths = Paths.Only("src/**");
			}));

			Job buildJob = Job.configure("build", UBUNTU_LATEST, job -> {
				job.steps.add(Checkout.configure());

				job.steps.add(UseAction.configure(
					"actions/setup-node@v4", "node", action ->
					    action.with.put("node-version", "20"))
				);

				job.steps.add(
					RunCommand.configure("npm run build && npm test")
				);
			});

			workflow.jobs.add(buildJob);

			workflow.jobs.add(Job.configure("deploy", UBUNTU_LATEST, job -> {
				job.needs.add(buildJob);
				job.steps.add(
					UseAction.configure("actions/deploy@v2", "deploy", action -> {
					    action.with.put("target", "production");
					    action.with.put("token", "${{ secrets.DEPLOY_TOKEN }}");
					}));
			}));
		});
	}
}

name: Deploy to Production

on:
  push:
	branches: [main]
	paths: ['src/**']

jobs:
  build:
	runs-on: ubuntu-22.04
	steps:
	  - uses: actions/checkout@v4
	  - uses: actions/setup-node@v4
		with:
		  node-version: '20'
	  - name: Build and Test
		run: npm run build && npm test

  deploy:
	needs: [build]
	runs-on: ubuntu-latest
	environment: production
	steps:
	  - uses: actions/deploy@v2
		with:
		  target: production
		  token: ${{ secrets.DEPLOY_TOKEN }}

Typeflows is a developer tool that runs locally, not a SaaS. It consists of:

• Language-specific DSLs (JVM (Kotlin/Java), TypeScript, Python +others) that you install via your package manager (Gradle, npm, pip)
• Build integrations that plug into your existing development workflow
• Community-supported package ecosystem allowing the import of secure best practices for GitHub configuration

Everything runs on developer machines and CI - no data is sent to external services. You maintain full control over your code and workflows with zero vendor lock-in.

While template tools create static, one-time snapshots, Typeflows provides living, evolving configuration management:

Template Tools:
• Generate files once, then you're on your own
• No version control for updates or bug fixes
• Manual copy/paste to get template changes
• No testing or validation of generated configs

Typeflows:
• Versioned packages with semantic versioning and changelog
• Automatic updates when you upgrade package versions
• Local testing to catch configuration errors before deployment
• Type safety prevents invalid GitHub configurations
• Composable libraries - mix multiple packages together

Think of template tools as scaffolding (useful for initial setup) vs. Typeflows as package management (ongoing maintenance and evolution). You wouldn't copy/paste your npm dependencies - why copy/paste your GitHub configurations?
